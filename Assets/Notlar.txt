		KATMAN YOKSAYMAK (LÝNECAST ÝÇÝN)		"~" bunun adý ters iþlemidir

	Bir katmaný yoksaymak için

var mask = 1 << 8;
mask = ~mask;

RaycastHit2D checkWall = Physics2D.Linecast(transform.position, target.position + new Vector3(0, 0.3f, 0), mask);

	
	
	Birden fazla katmaný yoksaymak için

var layer1 = 3;
var layer2 = 5;
var layermask1 = 1 << layer1;
var layermask2 = 1 << layer2;
var finalmask = ~(layermask1 | layermask2); // Or, (1 << layer1) | (1 << layer2)

RaycastHit2D checkWall = Physics2D.Linecast(transform.position, target.position + new Vector3(0, 0.3f, 0), finalmask);
